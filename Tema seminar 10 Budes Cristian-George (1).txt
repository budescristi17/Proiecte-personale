SELECT C.ID_COMANDA, C.DATA, SUM(R.CANTITATE * R.PRET) AS VALOARE_TOTALA
FROM COMENZI C
JOIN RAND_COMENZI R ON C.ID_COMANDA = R.ID_COMANDA
WHERE TO_CHAR(C.DATA, 'YYYY') = '1999'
GROUP BY C.ID_COMANDA, C.DATA
HAVING SUM(R.CANTITATE * R.PRET) > 2000
ORDER BY C.DATA;

--2
SELECT C.ID_COMANDA, C.DATA, 
       SUM(R.CANTITATE * R.PRET) AS VALOARE_TOTALA,
       CASE 
           WHEN SUM(R.CANTITATE * R.PRET) <= 1000 THEN 100
           WHEN SUM(R.CANTITATE * R.PRET) > 1000 AND SUM(R.CANTITATE * R.PRET) <= 2000 THEN 200
           ELSE 300 
       END AS TAXA_TRANSPORT
FROM COMENZI C
JOIN RAND_COMENZI R ON C.ID_COMANDA = R.ID_COMANDA
GROUP BY C.ID_COMANDA, C.DATA;

--3
SELECT C.ID_CLIENT, R.ID_PRODUS, SUM(R.CANTITATE) AS CANTITATE_TOTALA
FROM COMENZI C
JOIN RAND_COMENZI R ON C.ID_COMANDA = R.ID_COMANDA
GROUP BY C.ID_CLIENT, R.ID_PRODUS
ORDER BY C.ID_CLIENT;

FROM ANGAJATI A
JOIN DEPARTAMENTE D ON A.ID_DEPARTAMENT = D.ID_DEPARTAMENT
WHERE A.FUNCTIE = (SELECT FUNCTIE FROM ANGAJATI WHERE NUME = 'Russell')
  AND A.ANGAJAT_ID IN (SELECT DISTINCT ID_SEF FROM ANGAJATI WHERE ID_SEF IS NOT NULL);

FROM DEPARTAMENTE D
WHERE D.ID_DEPARTAMENT IN (SELECT ID_DEPARTAMENT FROM ANGAJATI)
MINUS
SELECT D.ID_DEPARTAMENT, D.DENUMIRE_DEPARTAMENT
FROM DEPARTAMENTE D
JOIN ANGAJATI A ON D.ID_DEPARTAMENT = A.ID_DEPARTAMENT
WHERE A.SALARIUL >= 10000;

FROM ANGAJATI 
WHERE FUNCTIE IN ('Programmer', 'Accountant') 
  AND TO_CHAR(DATA_ANGAJARII, 'YYYY') BETWEEN '1995' AND '1998';

UPDATE RAND_COMENZI
SET PRET = PRET * 1.15
WHERE PRET < (
    SELECT AVG(PRET_LISTA) 
    FROM PRODUSE 
    WHERE LOWER(NUME_PRODUS) LIKE '%sound%'
);
ROLLBACK;

--3
SELECT NUME, DATA_ANGAJARII, FUNCTIE AS FUNCTIE_ACTUALA
FROM ANGAJATI
WHERE ID_DEPARTAMENT IN (50, 80)
  AND TO_CHAR(DATA_ANGAJARII, 'YYYY') BETWEEN '1995' AND '1999';

--4
UPDATE RAND_COMENZI
SET PRET = PRET * 0.95
WHERE CANTITATE < (
    SELECT AVG(R.CANTITATE)
    FROM RAND_COMENZI R
    JOIN PRODUSE P ON R.ID_PRODUS = P.PRODUS_ID
    WHERE P.NUME_PRODUS LIKE '%CPU%'
);
ROLLBACK;

SELECT P.NUME_PRODUS, SUM(R.CANTITATE * R.PRET) AS VALOARE_TOTALA
FROM PRODUSE P
JOIN RAND_COMENZI R ON P.PRODUS_ID = R.ID_PRODUS
GROUP BY P.NUME_PRODUS
HAVING SUM(R.CANTITATE * R.PRET) BETWEEN 1500 AND 4000;

SELECT C.ID_COMANDA, C.DATA, SUM(R.CANTITATE * R.PRET) AS VALOARE_TOTALA, COUNT(R.ID_PRODUS) AS NR_PRODUSE
FROM COMENZI C
JOIN RAND_COMENZI R ON C.ID_COMANDA = R.ID_COMANDA
WHERE LOWER(C.MODALITATE) = 'online'
  AND TO_CHAR(C.DATA, 'YYYY') BETWEEN '1999' AND '2000'
GROUP BY C.ID_COMANDA, C.DATA
HAVING COUNT(R.ID_PRODUS) >= 2;

SELECT C.ID_COMANDA, P.NUME_PRODUS, R.PRET, R.CANTITATE, (R.PRET * R.CANTITATE) AS VALOARE
FROM COMENZI C
JOIN RAND_COMENZI R ON C.ID_COMANDA = R.ID_COMANDA
JOIN PRODUSE P ON R.ID_PRODUS = P.PRODUS_ID
WHERE LOWER(P.NUME_PRODUS) LIKE '%monitor%';

SELECT A.NUME, A.ID_DEPARTAMENT, COUNT(C.ID_COMANDA) AS NR_COMENZI
FROM ANGAJATI A
JOIN COMENZI C ON A.ANGAJAT_ID = C.ID_ANGAJAT
WHERE A.ID_DEPARTAMENT = 80
  AND TO_CHAR(C.DATA, 'MM') = '11'
GROUP BY A.NUME, A.ID_DEPARTAMENT;

UPDATE RAND_COMENZI
SET PRET = PRET * 0.90
WHERE PRET > (
    SELECT AVG(PRET) 
    FROM RAND_COMENZI 
    WHERE ID_PRODUS = 3155
);
ROLLBACK;

SELECT A.NUME, A.FUNCTIE, COUNT(C.ID_COMANDA) AS NR_COMENZI
FROM ANGAJATI A
JOIN COMENZI C ON A.ANGAJAT_ID = C.ID_ANGAJAT
GROUP BY A.NUME, A.FUNCTIE
HAVING COUNT(C.ID_COMANDA) >= 2;

UPDATE RAND_COMENZI
SET PRET = PRET * 1.05
WHERE (CANTITATE * PRET) >= 1000;
ROLLBACK;

SELECT A.NUME, COUNT(C.ID_COMANDA) AS NR_COMENZI, A.SALARIUL,
       CASE 
           WHEN COUNT(C.ID_COMANDA) BETWEEN 1 AND 2 THEN A.SALARIUL * 0.05
           WHEN COUNT(C.ID_COMANDA) BETWEEN 3 AND 5 THEN A.SALARIUL * 0.07
           WHEN COUNT(C.ID_COMANDA) > 5 THEN A.SALARIUL * 0.10
           ELSE 0
       END AS BONUS
FROM ANGAJATI A
JOIN COMENZI C ON A.ANGAJAT_ID = C.ID_ANGAJAT
GROUP BY A.NUME, A.SALARIUL;
