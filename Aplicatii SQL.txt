SELECT * FROM ANGAJATI;

-- SA SE AFISEZE PENTRU FIECARE ANGAJAT, NUMELE SI PRENUMELE SAU PRECUM SI NUMELE SI
-- PRENUMELE MANAGERULUI DIRECT (SELF JOIN)

SELECT A.ID_ANGAJAT,A.PRENUME,A.NUME,B.ID_ANGAJAT ID_SEF,B.PRENUME PRENUME_SEF,B.NUME NUME_SEF 
FROM ANGAJATI A LEFT JOIN ANGAJATI B 
ON A.ID_MANAGER=B.ID_ANGAJAT;

-- SA SE AFISEZE TOTI SUBORDONATII DIRECTI SI INDIRECTI AI LUI 101

SELECT ID_ANGAJAT,LEVEL,LPAD(' ',2*LEVEL-2)||NUME||' '||PRENUME ANGAJAT,SALARIUL
FROM ANGAJATI A
START WITH ID_ANGAJAT=101
CONNECT BY PRIOR ID_ANGAJAT=ID_MANAGER
ORDER SIBLINGS BY SALARIUL;


-- SA SE AFISEZE TOTI SUBORDONATII DIRECTI SI INDIRECTI AI LUI 100 CARE AU INTERMEDIAT CEL PUTIN 
-- O COMANDA

SELECT ID_ANGAJAT,LEVEL,LPAD(' ',2*LEVEL-2)||NUME||' '||PRENUME ANGAJAT,SALARIUL
FROM ANGAJATI A
WHERE ID_ANGAJAT IN (SELECT ID_ANGAJAT FROM COMENZI)
START WITH ID_ANGAJAT=100
CONNECT BY PRIOR ID_ANGAJAT=ID_MANAGER
ORDER SIBLINGS BY SALARIUL;

-- SA SE AFISEZE TOTI SUBORDONATII DIRECTI SI INDIRECTI AI LUI 100 MAI PUTIN 
-- ANGAJATUL 101 SI SUBORDONATII SAI


SELECT ID_ANGAJAT,LEVEL,LPAD(' ',2*LEVEL-2)||NUME||' '||PRENUME ANGAJAT,SALARIUL
FROM ANGAJATI A
START WITH ID_ANGAJAT=100
CONNECT BY PRIOR ID_ANGAJAT=ID_MANAGER
MINUS
SELECT ID_ANGAJAT,LEVEL,LPAD(' ',2*LEVEL-2)||NUME||' '||PRENUME ANGAJAT,SALARIUL
FROM ANGAJATI A
START WITH ID_ANGAJAT=101
CONNECT BY PRIOR ID_ANGAJAT=ID_MANAGER
;


SELECT ID_ANGAJAT,LEVEL,LPAD(' ',2*LEVEL-2)||NUME||' '||PRENUME ANGAJAT,SALARIUL
FROM ANGAJATI A 
START WITH ID_ANGAJAT=100 
CONNECT BY PRIOR ID_ANGAJAT=ID_MANAGER AND ID_ANGAJAT<>101;

-- SA SE AFISEZE TOTI SUBORDONATII DIRECTI SI INDIRECTI AI LUI Kochhar Neena

SELECT ID_ANGAJAT,LEVEL,LPAD(' ',2*LEVEL-2)||NUME||' '||PRENUME ANGAJAT,SALARIUL
FROM ANGAJATI A
START WITH NUME='Kochhar' AND PRENUME='Neena'
CONNECT BY PRIOR ID_ANGAJAT=ID_MANAGER
ORDER SIBLINGS BY SALARIUL;


-- SA SE AFISEZE TOTI SUBORDONATII DIRECTI SI INDIRECTI AI ANG DIN DEP 90

SELECT ID_ANGAJAT,LEVEL,LPAD(' ',2*LEVEL-2)||NUME||' '||PRENUME ANGAJAT,SALARIUL,ID_DEPARTAMENT
FROM ANGAJATI A
START WITH ID_DEPARTAMENT=90
CONNECT BY PRIOR ID_ANGAJAT=ID_MANAGER
ORDER SIBLINGS BY SALARIUL;

-- SA SE AFISEZE TOTI SUPERIORII DIRECTI SI INDIRECTI AI LUI 110 (CHEN JOHN)
SELECT ID_ANGAJAT,LEVEL,LPAD(' ',2*LEVEL-2)||NUME||' '||PRENUME ANGAJAT,SALARIUL
FROM ANGAJATI A
START WITH ID_ANGAJAT=110
CONNECT BY ID_ANGAJAT=PRIOR ID_MANAGER
ORDER SIBLINGS BY SALARIUL;


-- SA SE AFISEZE TOTI SUBORDONATII DIRECTI SI INDIRECTI AI LUI 101. SE VA AFISA CALEA CATRE
-- FIECARE ANGAJAT

SELECT ID_ANGAJAT,LEVEL,LPAD(' ',2*LEVEL-2)||NUME||' '||PRENUME ANGAJAT,SALARIUL,
SYS_CONNECT_BY_PATH(NUME,'/') CALE,connect_by_root nume  sef
FROM ANGAJATI A
START WITH ID_DEPARTAMENT=90
CONNECT BY PRIOR ID_ANGAJAT=ID_MANAGER
ORDER SIBLINGS BY SALARIUL;

SELECT * FROM ANGAJATI WHERE ID_DEPARTAMENT=90;

--Sa se afiseze cati subordonati are angajatul 100 pe fiecare nivel
SELECT LEVEL,count(*) nr
FROM angajati
where level>1
START WITH id_angajat=100
CONNECT BY PRIOR id_angajat = id_manager
group by LEVEL
order by 1;

-- SA SE CONSTRUIASCA O TABELA VIRTUALA CARE AFISEAZA TOTI SUBORDONATII DIRECTI 
-- SI INDIRECTI AI LUI 101. SE VA AFISA CALEA CATRE FIECARE ANGAJAT
-- OBS: UN VIEW OBISNUIT NU STOCHEAZA DATE, EL DOAR REFERA DATELE DIN TABELA SAU TABELELE DE BAZA
CREATE VIEW V_SUBORD_101 AS
SELECT ID_ANGAJAT,LEVEL NIVEL,LPAD(' ',2*LEVEL-2)||NUME||' '||PRENUME ANGAJAT,SALARIUL,
SYS_CONNECT_BY_PATH(NUME,'/') CALE,connect_by_root nume  sef
FROM ANGAJATI A
START WITH ID_DEPARTAMENT=90
CONNECT BY PRIOR ID_ANGAJAT=ID_MANAGER
ORDER SIBLINGS BY SALARIUL;

SELECT * FROM V_SUBORD_101 WHERE SALARIUL>10000;
