#!/bin/bash

clear
echo -n "Introduceti valoarea lui n: "
read n

fibonacci()
{
    if [ $1 -le 0 ]; 
    then
        echo 0
    elif [ $1 -eq 1 ]; 
    then
        echo 1
    else
        n1=$(fibonacci $(( $1 - 1 )))
        n2=$(fibonacci $(( $1 - 2 )))
        echo $(( n1 + n2 ))
    fi
}

rezultat=$(fibonacci $n)
echo "Termenul al $n-lea din sirul Fibonacci este: $rezultat"
