// ---------------- CLASA ABSTRACTA MARFA ----------------
class Marfa {
protected:
    string cod;
    string nume;
    double pretUnitar;
    int anFabricatie;
    string tara;
    double indiceCalitate; // 1.0 - 10.0

public:
    // Constructor cu parametri si valori implicite
    Marfa(string c = "", string n = "", double p = 0, int an = 0, string t = "", double ic = 1.0)
        : cod(c), nume(n), pretUnitar(p), anFabricatie(an), tara(t), indiceCalitate(ic) {}

    virtual ~Marfa() {}

    // Getteri necesari
    string getCod() const { return cod; }
    string getTara() const { return tara; }
    double getIndiceCalitate() const { return indiceCalitate; }
    string getNume() const { return nume; }

    // Metode virtuale
    virtual string getTipTransport() = 0; // Metoda pur virtuala
    virtual double getPretFinalTransport() { return pretUnitar; }

    // Afisare
    virtual void afisare(ostream& out) const {
        out << left << setw(6) << cod << setw(20) << nume << setw(15) << tara 
            << "Pret:" << setw(8) << pretUnitar << "Calitate:" << indiceCalitate;
    }

    friend ostream& operator<<(ostream& out, const Marfa& m) {
        m.afisare(out);
        return out;
    }
};

// ---------------- CLASE DERIVATE ----------------

class MarfaGenerala : public Marfa {
protected:
    double greutate;
    double volum;

public:
    MarfaGenerala(string c="", string n="", double p=0, int an=0, string t="", double ic=1.0, double g=0, double v=0)
        : Marfa(c, n, p, an, t, ic), greutate(g), volum(v) {}

    string getTipTransport() override { return "naval"; }

    double getPretFinalTransport() override {
        double pret = pretUnitar;
        if (greutate > 1000) pret += pret * 0.20; // +20% taxa
        return pret;
    }

    void afisare(ostream& out) const override {
        Marfa::afisare(out);
        out << " | Naval (G:" << greutate << ")";
    }
};

class MarfaPerisabila : public Marfa {
    double tempDepozitare;
    int durataValabilitate;

public:
    MarfaPerisabila(string c="", string n="", double p=0, int an=0, string t="", double ic=1.0, double temp=0, int dur=0)
        : Marfa(c, n, p, an, t, ic), tempDepozitare(temp), durataValabilitate(dur) {}

    string getTipTransport() override { return "aerian"; }

    double getPretFinalTransport() override {
        return pretUnitar + (pretUnitar * 0.50); // +50% taxa refrigerare
    }

    void afisare(ostream& out) const override {
        Marfa::afisare(out);
        out << " | Aerian (Temp:" << tempDepozitare << ")";
    }
};

class MarfaDigitala : public MarfaGenerala {
    string codLicenta;

public:
    // Mosteneste MarfaGenerala, greutate/volum sunt 0
    MarfaDigitala(string c="", string n="", double p=0, int an=0, string t="", double ic=1.0, string lic="")
        : MarfaGenerala(c, n, p, an, t, ic, 0, 0), codLicenta(lic) {}

    string getTipTransport() override { return "electronic"; }

    double getPretFinalTransport() override {
        return pretUnitar - (pretUnitar * 0.40); // -40% reducere
    }

    void afisare(ostream& out) const override {
        Marfa::afisare(out);
        out << " | Electronic (Lic:" << codLicenta << ")";
    }
};

// ---------------- CLASA FIRMA ----------------
class Firma {
    vector<Marfa*> inventar;
    double profitTotal = 0;
    map<string, int> tranzactiiPeTari;

public:
    ~Firma() {
        for (auto m : inventar) delete m;
    }

    void adaugaMarfa(Marfa* m) {
        for (auto existent : inventar) {
            if (existent->getCod() == m->getCod()) {
                return; // Cod unic, nu adaugam duplicate
            }
        }
        inventar.push_back(m);
    }

    friend ostream& operator<<(ostream& out, const Firma& f) {
        out << "\n--- Inventar Firma ---\n";
        for (auto m : f.inventar) {
            out << *m << endl;
        }
        return out;
    }

    void exporta(string cod, int cantitate) {
        for (auto m : inventar) {
            if (m->getCod() == cod) {
                double valoare = m->getPretFinalTransport() * cantitate;
                profitTotal += valoare;
                tranzactiiPeTari[m->getTara()] += cantitate;
                return;
            }
        }
    }

    // Operator /= : returneaza marfuri cu pret <= pretMaxim
    vector<Marfa*> operator/=(double pretMaxim) {
        vector<Marfa*> rezultat;
        for (auto m : inventar) {
            if (m->getPretFinalTransport() <= pretMaxim) {
                rezultat.push_back(m);
            }
        }
        return rezultat;
    }

    // Metoda principala de logica din examen
    void genereazaRaportCalitatePiete(vector<string> coduriExportate) {
        map<string, int> frecventaCoduri;
        for (const string& cod : coduriExportate) {
            frecventaCoduri[cod]++;
        }

        // Pas 1: Calculam media indicilor de calitate pe fiecare tara
        map<string, pair<double, int>> statisticaTari; // Tara -> <SumaIndici, NumarProduse>
        
        for (auto m : inventar) {
            statisticaTari[m->getTara()].first += m->getIndiceCalitate();
            statisticaTari[m->getTara()].second++;
        }

        cout << "\nRezultat raport calitate piete (Coduri acceptate):\n";
        bool first = true;

        // Pas 2: Iteram prin codurile unice exportate (map-ul le sorteaza automat alfabetic)
        for (auto const& [cod, count] : frecventaCoduri) {
            // Cautam produsul in inventar
            Marfa* produs = nullptr;
            for (auto m : inventar) {
                if (m->getCod() == cod) {
                    produs = m;
                    break;
                }
            }

            if (produs) {
                string tara = produs->getTara();
                double medieTara = statisticaTari[tara].first / statisticaTari[tara].second;

                // Verificam conditia: indice produs >= medie tara
                if (produs->getIndiceCalitate() >= medieTara) {
                    if (!first) cout << ", ";
                    cout << cod;
                    first = false;
                }
            }
        }
        cout << endl;
    }
};
